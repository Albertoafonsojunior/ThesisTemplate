% Macro for embedding papers
\newenvironment{paper}[2]
{
  \def\paperpath{#1}
  \chapter*{\papercite{#2}}
  \pagecolor{whiteish}
  \noindent\\[-2ex]
  \fullpapercite{#2}\\[2ex]
  \noindent
  \textbf{Contribution}\\
}
{
  \cleardoublepage
  \nopagecolor
  \color{black}
  \includepdf[pages={-}]{\paperpath}
  \cleardoublepage
}
